import { getWeatherByDate } from "/common/weather/weather.js"; async function showWeather(date) { const w = await getWeatherByDate(date); const box = document.getElementById("weatherBox"); if (!w) { box.innerHTML = "データなし"; return; } box.innerHTML = `

最高気温：${w.tmax} ℃

最低気温：${w.tmin} ℃

降水量：${w.precip} mm

日照時間：${w.sunshine} h

`; }
